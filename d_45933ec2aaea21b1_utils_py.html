26def final_step_like( 

27 step: base.Step, next_observations: types.NestedArray, next_extras: dict = None 

28) -> base.Step: 

29 """Return a list of steps with the final step zero-filled.""" 

30 # Make zero-filled components so we can fill out the last step. 

31 zero_action, zero_reward, zero_discount = tree.map_structure( 

32 acme_utils.zeros_like, (step.actions, step.rewards, step.discounts) 

33 ) 

34 

35 return base.Step( 

36 observations=next_observations, 

37 actions=zero_action, 

38 rewards=zero_reward, 

39 discounts=zero_discount, 

40 start_of_episode=False, 

41 extras=next_extras 

42 if next_extras 

43 else tree.map_structure(acme_utils.zeros_like, step.extras), 

44 )

47def trajectory_signature( 

48 environment_spec: specs.EnvironmentSpec, 

49 sequence_length: Optional[int] = None, 

50 extras_spec: types.NestedSpec = (), 

51) -> tf.TypeSpec: 

52 """This is a helper method for generating signatures for Reverb tables. 

53 

54 Signatures are useful for validating data types and shapes, see Reverb's 

55 documentation for details on how they are used. 

56 

57 Args: 

58 environment_spec: A `specs.EnvironmentSpec` whose fields are nested 

59 structures with leaf nodes that have `.shape` and `.dtype` attributes. 

60 This should come from the environment that will be used to generate 

61 the data inserted into the Reverb table. 

62 extras_spec: A nested structure with leaf nodes that have `.shape` and 

63 `.dtype` attributes. The structure (and shapes/dtypes) of this must 

64 be the same as the `extras` passed into `ReverbAdder.add`. 

65 sequence_length: An optional integer representing the expected length of 

66 sequences that will be added to replay. 

67 

68 Returns: 

69 A `Trajectory` whose leaf nodes are `tf.TensorSpec` objects. 

70 """ 

71 

72 def add_time_dim(paths: Iterable[str], spec: tf.TensorSpec) -> None: 

73 return tf.TensorSpec( 

74 shape=(sequence_length, *spec.shape), 

75 dtype=spec.dtype, 

76 name="/".join(str(p) for p in paths), 

77 ) 

78 

79 agent_specs = environment_spec.get_agent_specs() 

80 agents = environment_spec.get_agent_ids() 

81 env_extras_spec = environment_spec.get_extra_specs() 

82 extras_spec.update(env_extras_spec) 

83 

84 obs_specs = {} 

85 act_specs = {} 

86 reward_specs = {} 

87 step_discount_specs = {} 

88 for agent in agents: 

89 rewards_spec, step_discounts_spec = tree_utils.broadcast_structures( 

90 agent_specs[agent].rewards, agent_specs[agent].discounts 

91 ) 

92 obs_specs[agent] = agent_specs[agent].observations 

93 act_specs[agent] = agent_specs[agent].actions 

94 reward_specs[agent] = rewards_spec 

95 step_discount_specs[agent] = step_discounts_spec 

96 

97 # Add a time dimension to the specs 

98 ( 

99 obs_specs, 

100 act_specs, 

101 reward_specs, 

102 step_discount_specs, 

103 soe_spec, 

104 extras_spec, 

105 ) = tree.map_structure_with_path( 

106 add_time_dim, 

107 ( 

108 obs_specs, 

109 act_specs, 

110 reward_specs, 

111 step_discount_specs, 

112 specs.Array(shape=(), dtype=bool), 

113 extras_spec, 

114 ), 

115 ) 

116 

117 spec_step = base.Trajectory( 

118 observations=obs_specs, 

119 actions=act_specs, 

120 rewards=reward_specs, 

121 discounts=step_discount_specs, 

122 start_of_episode=soe_spec, 

123 extras=extras_spec, 

124 ) 

125 

126 return spec_step
