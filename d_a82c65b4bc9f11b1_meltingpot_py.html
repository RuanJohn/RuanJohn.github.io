49class MeltingpotEnvWrapper(ParallelEnvWrapper): 

50 """Environment wrapper for Melting pot.""" 

51 

52 def __init__( 

53 self, 

54 environment: Union[Substrate, Scenario], 

55 preprocessor: Callable[[Dict[str, NestedArray]], np.ndarray] = obs_preprocessor, 

56 ): 

57 """Constructor for Melting pot wrapper. 

58 Args: 

59 environment (Substrate or Scenario): Melting pot substrate or scenario. 

60 preprocessor (Callable[[Dict[str, NestedArray]], np.ndarray]): function that 

61 transforms an observation to a single array 

62 """ 

63 self._environment = environment 

64 self._reset_next_step = True 

65 self._env_done = False 

66 self._num_agents = len(self._environment.action_spec()) 

67 self._num_actions = self._environment.action_spec()[0].num_values 

68 self._env_image: Optional[np.ndarray] = None 

69 self._screen = None 

70 self._preprocessor = preprocessor 

71 

72 # individual agent obervation 

73 _, _, _, obs = self._environment.reset() 

74 ob = self._preprocessor(obs[0]) 

75 self._ob_spec = specs.Array( 

76 shape=ob.shape, dtype=ob.dtype, name="indv_agent_ob" 

77 )

79 def reset(self) -> dm_env.TimeStep: 

80 """Resets the env. 

81 Returns: 

82 dm_env.TimeStep: dm timestep. 

83 """ 

84 

85 timestep = self._environment.reset() 

86 self._reset_next_step = False 

87 

88 self._set_env_image() 

89 

90 return self._refine_timestep(timestep)

98 def _to_olt( 

99 self, observation: Dict[str, NestedArray], num_values: int, is_terminal: bool 

100 ) -> types.OLT: 

101 """Createa an OLT from a observation. 

102 It just computes the legal actions and terminal. All actions are legal and 

103 terminal is determined with timestep.last() 

104 Args: 

105 observation (TimeStep): the observation 

106 num_values (int): the number of actions 

107 is_terminal (bool): whether its a terminal observation 

108 Returns: 

109 types.OLT: observation, legal actions, and terminal 

110 """ 

111 legal_actions = np.ones([num_values], dtype=np.int32) 

112 terminal = np.asarray([is_terminal], dtype=np.float32) 

113 return types.OLT( 

114 observation=self._preprocessor(observation), 

115 legal_actions=legal_actions, 

116 terminal=terminal, 

117 )

119 def _to_dict_observation( 

120 self, observation: List[Dict[str, NestedArray]], is_terminal: bool 

121 ) -> Dict[str, types.OLT]: 

122 """Observation list to dict 

123 Transforms a list of observations into a dictionary of observations 

124 with keys corresponding to the agent ids 

125 Args: 

126 observation (List[Dict[str, NestedArray]]): List observation 

127 is_terminal (bool): whether the observations corresponds to a 

128 terminal timestep 

129 Returns: 

130 Dict[str, types.OLT]: Dictionary of observations 

131 """ 

132 return { 

133 f"agent_{i}": self._to_olt(obs, self._num_actions, is_terminal) 

134 for i, obs in enumerate(observation) 

135 }

137 def _to_dict_rewards(self, rewards: List[NestedArray]) -> Dict[str, NestedArray]: 

138 """List of rewards to Dict of rewards 

139 Transforms a list of rewards to a dictionary of rewards with keys corresponding 

140 to the agent ids 

141 Args: 

142 rewards (List[NestedArray]): List of rewards 

143 Returns: 

144 Dict[str, NestedArray]: Dictionary of reward 

145 """ 

146 return { 

147 f"agent_{i}": np.dtype("float32").type(rew) for i, rew in enumerate(rewards) 

148 }

150 def _to_dict_discounts( 

151 self, discounts: List[NestedArray] 

152 ) -> Dict[str, NestedArray]: 

153 """List of dicounts to Dict of discounts 

154 Transforms a list of discounts into a dictionary of discounts with keys 

155 corresponding to the agent ids 

156 Args: 

157 discounts (List[NestedArray]): List of discounts 

158 Returns: 

159 Dict[str, NestedArray]: Dictionary of discounts 

160 """ 

161 return { 

162 f"agent_{i}": np.dtype("float32").type(disc) 

163 for i, disc in enumerate(discounts) 

164 }

166 def _refine_timestep(self, timestep: dm_env.TimeStep) -> dm_env.TimeStep: 

167 """Converts a melting pot timestep into one compatiple with Mava 

168 The difference between the timestep from melting pot and that of Mava is 

169 that for the observation, reward, and discount, mava expects dictionaries 

170 with keys corresponding to the agent ids while melting pot simply uses a 

171 list for this. 

172 Args: 

173 timestep (dm_env.TimeStep): a timestep from melting pot 

174 Returns: 

175 dm_env.TimeStep: a timestep compatible with Mava 

176 """ 

177 is_terminal = timestep.last() 

178 observation = self._to_dict_observation(timestep.observation, is_terminal) 

179 reward = self._to_dict_rewards(timestep.reward) 

180 discount = self._to_dict_discounts([timestep.discount] * self._num_agents) 

181 return dm_env.TimeStep(timestep.step_type, reward, discount, observation)

183 def step(self, actions: Dict[str, np.ndarray]) -> dm_env.TimeStep: 

184 """Steps in env. 

185 Args: 

186 actions (Dict[str, np.ndarray]): actions per agent. 

187 Returns: 

188 dm_env.TimeStep: dm timestep 

189 """ 

190 

191 if self._reset_next_step: 

192 return self.reset() 

193 

194 actions_ = [actions[f"agent_{i}"] for i in range(self._num_agents)] 

195 timestep = self._environment.step(actions_) 

196 self._set_env_image() 

197 timestep = self._refine_timestep(timestep) 

198 

199 if timestep.last(): 

200 self._reset_next_step = True 

201 self._env_done = True 

202 

203 return timestep

205 def render( 

206 self, mode: str = "human", screen_width: int = 800, screen_height: int = 600 

207 ) -> Optional[np.ndarray]: 

208 """Renders the environment in a pygame window or returns an image 

209 Args: 

210 mode (str, optional): mode for the display either rgb_array or human. 

211 Defaults to "human". 

212 screen_width (int, optional): the screen width. Defaults to 800. 

213 screen_height (int, optional): the screen height. Defaults to 600. 

214 Raises: 

215 ValueError: for invalid mode 

216 Returns: 

217 [np.ndarray]: an image array for mode, rgb_array 

218 """ 

219 if self._env_image: 

220 image = self._env_image 

221 height, width, _ = image.shape 

222 scale = min(screen_height // height, screen_width // width) 

223 if mode == "human": 

224 os.environ["PYGAME_HIDE_SUPPORT_PROMPT"] = "hide" 

225 if self._screen is None: 

226 pygame.init() 

227 self._screen = pygame.display.set_mode( # type: ignore 

228 (screen_width * scale, screen_height * scale) 

229 ) 

230 image = np.transpose(image, (1, 0, 2)) # PyGame is column major! 

231 surface = pygame.surfarray.make_surface(image) 

232 rect = surface.get_rect() 

233 surf = pygame.transform.scale( 

234 surface, (rect[2] * scale, rect[3] * scale) 

235 ) 

236 self._screen.blit(surf, dest=(0, 0)) # type: ignore 

237 pygame.display.update() 

238 return None 

239 elif mode == "rgb_array": 

240 return image 

241 else: 

242 raise ValueError("bad value for render mode") 

243 return None
