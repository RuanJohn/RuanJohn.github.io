279 def trainer( 

280 self, 

281 replay: reverb.Client, 

282 counter: counting.Counter, 

283 ) -> mava.core.Trainer: 

284 """System trainer 

285 

286 Args: 

287 replay (reverb.Client): replay data table to pull data from. 

288 counter (counting.Counter): step counter object. 

289 

290 Raises: 

291 Exception: no option for recurrence (yet). 

292 

293 Returns: 

294 mava.core.Trainer: system trainer. 

295 """ 

296 

297 # Create the networks to optimize (online) 

298 networks = self._network_factory( # type: ignore 

299 environment_spec=self._environment_spec, 

300 agent_net_keys=self._agent_net_keys, 

301 ) 

302 

303 # Create system architecture 

304 architecture = self._architecture( 

305 environment_spec=self._environment_spec, 

306 value_networks=networks["q_networks"], 

307 agent_net_keys=self._agent_net_keys, 

308 ) 

309 

310 # Fingerprint module 

311 if self._builder._replay_stabiliser_fn is not None: 

312 architecture = self._builder._replay_stabiliser_fn( # type: ignore 

313 architecture 

314 ) 

315 

316 # Communication module 

317 # NOTE: this is currently not expected to work with qmix 

318 # since we do not have a recurrent version. 

319 if self._communication_module_fn is not None: 

320 raise Exception( 

321 "QMIX currently does not support recurrence and \ 

322 therefore cannot use a communication module." 

323 ) 

324 

325 # Mixing module 

326 system_networks = self._mixer( 

327 environment_spec=self._environment_spec, 

328 architecture=architecture, 

329 num_hypernet_layers=self._num_hypernet_layers, 

330 qmix_hidden_dim=self._qmix_hidden_dim, 

331 hypernet_hidden_dim=self._hypernet_hidden_dim, 

332 ).create_system() 

333 

334 # Create logger 

335 trainer_logger_config = {} 

336 if self._logger_config and "trainer" in self._logger_config: 

337 trainer_logger_config = self._logger_config["trainer"] 

338 trainer_logger = self._logger_factory( # type: ignore 

339 "trainer", **trainer_logger_config 

340 ) 

341 

342 dataset = self._builder.make_dataset_iterator(replay) 

343 counter = counting.Counter(counter, "trainer") 

344 

345 return self._builder.make_trainer( 

346 networks=system_networks, 

347 dataset=dataset, 

348 counter=counter, 

349 communication_module=None, 

350 logger=trainer_logger, 

351 )
