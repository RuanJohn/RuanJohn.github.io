118 def _forward(self, inputs: Any) -> None: 

119 data = tree.map_structure( 

120 lambda v: tf.expand_dims(v, axis=0) if len(v.shape) <= 1 else v, inputs.data 

121 ) 

122 data = tf2_utils.batch_to_sequence(data) 

123 

124 observations, actions, rewards, discounts, _, _ = ( 

125 data.observations, 

126 data.actions, 

127 data.rewards, 

128 data.discounts, 

129 data.start_of_episode, 

130 data.extras, 

131 ) 

132 

133 # Using extra directly from inputs due to shape. 

134 core_state = tree.map_structure( 

135 lambda s: s[:, 0, :], inputs.data.extras["core_states"] 

136 ) 

137 core_message = tree.map_structure( 

138 lambda s: s[:, 0, :], inputs.data.extras["core_messages"] 

139 ) 

140 T = actions[self._agents[0]].shape[0] 

141 

142 # Use fact that end of episode always has the reward to 

143 # find episode lengths. This is used to mask loss. 

144 ep_end = tf.argmax(tf.math.abs(rewards[self._agents[0]]), axis=0) 

145 

146 with tf.GradientTape(persistent=True) as tape: 

147 q_network_losses: Dict[str, NestedArray] = { 

148 agent: {"policy_loss": tf.zeros(())} for agent in self._agents 

149 } 

150 

151 state = {agent: core_state[agent][0] for agent in self._agents} 

152 target_state = {agent: core_state[agent][0] for agent in self._agents} 

153 

154 message = {agent: core_message[agent][0] for agent in self._agents} 

155 target_message = {agent: core_message[agent][0] for agent in self._agents} 

156 

157 # _target_q_networks must be 1 step ahead 

158 target_channel = self._communication_module.process_messages(target_message) 

159 for agent in self._agents: 

160 agent_key = self._agent_net_keys[agent] 

161 (q_targ, m), s = self._target_q_networks[agent_key]( 

162 observations[agent].observation[0], 

163 target_state[agent], 

164 target_channel[agent], 

165 ) 

166 target_state[agent] = s 

167 target_message[agent] = m 

168 

169 for t in range(1, T, 1): 

170 channel = self._communication_module.process_messages(message) 

171 target_channel = self._communication_module.process_messages( 

172 target_message 

173 ) 

174 

175 for agent in self._agents: 

176 agent_key = self._agent_net_keys[agent] 

177 

178 # Cast the additional discount 

179 # to match the environment discount dtype. 

180 

181 discount = tf.cast(self._discount, dtype=discounts[agent][0].dtype) 

182 

183 (q_targ, m), s = self._target_q_networks[agent_key]( 

184 observations[agent].observation[t], 

185 target_state[agent], 

186 target_channel[agent], 

187 ) 

188 

189 target_state[agent] = s 

190 target_message[agent] = tf.math.multiply( 

191 m, observations[agent].observation[t][:, :1] 

192 ) 

193 

194 (q, m), s = self._q_networks[agent_key]( 

195 observations[agent].observation[t - 1], 

196 state[agent], 

197 channel[agent], 

198 ) 

199 

200 state[agent] = s 

201 message[agent] = tf.math.multiply( 

202 m, observations[agent].observation[t - 1][:, :1] 

203 ) 

204 

205 # Mask target 

206 q_targ = tf.concat( 

207 [ 

208 [q_targ[i]] 

209 if t <= ep_end[i] 

210 else [tf.zeros_like(q_targ[i])] 

211 for i in range(q_targ.shape[0]) 

212 ], 

213 axis=0, 

214 ) 

215 

216 loss, _ = trfl.qlearning( 

217 q, 

218 actions[agent][t - 1], 

219 rewards[agent][t - 1], 

220 discount * discounts[agent][t], 

221 q_targ, 

222 ) 

223 

224 # Index loss (mask ended episodes) 

225 if not tf.reduce_any(t - 1 <= ep_end): 

226 continue 

227 

228 loss = tf.reduce_mean(loss[t - 1 <= ep_end]) 

229 # loss = tf.reduce_mean(loss) 

230 q_network_losses[agent]["policy_loss"] += loss 

231 

232 self._q_network_losses = q_network_losses 

233 self.tape = tape
